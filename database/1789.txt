1789. Primary Department for Each Employee

# Write your MySQL query statement below
select employee_id, department_id
from Employee
where 
(
    primary_flag = "N"
    and employee_id in (select employee_id from Employee group by employee_id having count(*) = 1)
)
or 
(
    primary_flag = "Y"
    and employee_id in (select employee_id from Employee group by employee_id having count(*) > 1)
)
